# SAFE examples (should NOT trigger)
def safe_parameterized_query():
    user_id = request.GET.get('id')
    query = "SELECT * FROM users WHERE id = ?"
    cursor.execute(query, (user_id,))  # SAFE: Parameterized query

def safe_text_content():
